  <title>👧 칭찬 도장판 예제</title>

</head>

<body>

  <h2>📚 Supabase - 학생 정보 저장하기</h2>


  <button onclick="saveStudent()">학생 저장하기</button>

  <button onclick="loadStudents()">학생 불러오기</button>


  <ul id="studentList"></ul>


  <script src="https://cdn.jsdelivr.net/npm/@supabase/supabase-js"></script>

  <script>

    // ✅ 여기에 본인 Supabase 정보 넣으세요!

    const supabaseUrl = 'https://YOUR_PROJECT.supabase.co';  // ← 수정

    const supabaseKey = 'YOUR_ANON_KEY';                    // ← 수정

    const supabase = supabase.createClient(supabaseUrl, supabaseKey);


    // 저장하기

    async function saveStudent() {

      const name = prompt("학생 이름은?");

      const score = parseInt(prompt("점수는?"));


      const { data, error } = await supabase

        .from('students')

        .insert([{ name, score }]);


      if (error) {

        alert("❌ 저장 실패: " + error.message);

      } else {

        alert("✅ 저장 성공!");

        loadStudents(); // 자동 새로고침

      }

    }


    // 불러오기

    async function loadStudents() {

      const { data, error } = await supabase

        .from('students')

        .select('*')

        .order('score', { ascending: false });


      const list = document.getElementById("studentList");

      list.innerHTML = '';


      if (data) {

        data.forEach((student) => {

          const li = document.createElement("li");

          li.innerText = `${student.name} - ${student.score}점`;

          list.appendChild(li);

        });

      }

    }


    // 첫 로딩 시 학생 목록 자동 불러오기

    loadStudents();

  </script>

</body>

</html>
